//함수의 형태와 특징들에 대해 알게 해주는 코드
#include <stdio.h>
int get_num(void); // 함수 선언
int main(void) 
{
	int result;
	result = get_num(); // 함수 호출, 반환값은 result에 저장
	printf ("%d", result); // 반환값 출력
	return 0;
}
int get_num(void) // 매개변수가 없고 반환형만 있다.
{
	int num; // 키보드 입력값을 저장할 변수
	scanf ("%d", &num); // 키보드 입력
	return num; // 입력한 값 반환
}
